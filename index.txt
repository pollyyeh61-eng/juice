    <div class="bg-gradient-to-r from-green-500 to-emerald-600 p-6 text-white rounded-b-[2rem] shadow-lg">
        <h1 class="text-2xl font-bold tracking-tight">活力美髮學堂 ✂️</h1>
        <p class="opacity-90 mt-1 text-sm">今日限定：現榨芭樂汁團購</p>
    </div>

    <div class="p-4 grid grid-cols-1 gap-4 mt-2" id="product-list">
        </div>

    <div id="bottom-bar" class="fixed bottom-6 left-4 right-4 bg-black/80 backdrop-blur-md text-white p-4 rounded-3xl shadow-2xl flex justify-between items-center transform translate-y-32 transition-transform duration-500 z-50">
        <div>
            <span class="text-gray-400 text-xs block">已選項目</span>
            <span class="font-bold text-xl text-green-400">$<span id="total-price">0</span></span>
        </div>
        <button onclick="openCheckout()" class="bg-green-500 hover:bg-green-400 text-black px-6 py-2 rounded-2xl font-black transition-colors">
            立即訂購
        </button>
    </div>

    <div id="drawer" class="fixed inset-0 bg-black/60 z-[100] hidden backdrop-blur-sm">
        <div class="absolute bottom-0 left-0 right-0 bg-white rounded-t-[2.5rem] p-8 animate-slide-up">
            <div class="w-12 h-1.5 bg-gray-200 rounded-full mx-auto mb-6"></div>
            <h2 class="text-2xl font-black mb-4 text-gray-800">最後確認</h2>
            
            <div id="order-summary" class="max-h-40 overflow-y-auto mb-6 space-y-2">
                </div>

            <div class="space-y-4">
                <input type="text" id="user-name" placeholder="您的姓名" class="w-full bg-gray-100 p-4 rounded-2xl outline-none focus:ring-2 focus:ring-green-500">
                <button onclick="finalizeOrder()" id="final-btn" class="w-full bg-green-500 text-white py-4 rounded-2xl font-bold text-lg shadow-xl shadow-green-200">
                    確認並傳送簡訊
                </button>
                <button onclick="closeCheckout()" class="w-full text-gray-400 py-2 text-sm">返回修改</button>
            </div>
        </div>
    </div>

    <script>
        // --- Firebase 設定 ---
        const firebaseConfig = {
            apiKey: "你的API_KEY",
            databaseURL: "你的資料庫網址",
            projectId: "你的專案ID",
            appId: "你的APP_ID"
        };
        firebase.initializeApp(firebaseConfig);
        const db = firebase.database();

        const products = [
            { id: 'g1', name: '原味芭樂汁', price: 50, emoji: '🟢' },
            { id: 'g2', name: '甘梅芭樂汁', price: 55, emoji: '🔴' },
            { id: 'g3', name: '檸檬芭樂汁', price: 60, emoji: '🟡' }
        ];

        let cart = {};

        function renderProducts() {
            const list = document.getElementById('product-list');
            products.forEach(p => {
                list.innerHTML += `
                    <div class="product-card bg-white p-5 rounded-3xl shadow-sm flex justify-between items-center border border-gray-100 transition-all">
                        <div class="flex items-center gap-4">
                            <div class="text-3xl bg-gray-50 w-14 h-14 flex items-center justify-center rounded-2xl">${p.emoji}</div>
                            <div>
                                <h3 class="font-bold text-gray-800">${p.name}</h3>
                                <p class="text-green-600 font-bold">$${p.price}</p>
                            </div>
                        </div>
                        <div class="flex items-center gap-3 bg-gray-50 rounded-2xl p-1">
                            <button onclick="updateCart('${p.id}', -1)" class="w-10 h-10 flex items-center justify-center text-xl font-bold text-gray-400">-</button>
                            <span id="qty-${p.id}" class="w-6 text-center font-bold text-gray-700">0</span>
                            <button onclick="updateCart('${p.id}', 1)" class="w-10 h-10 flex items-center justify-center text-xl font-bold text-green-600">+</button>
                        </div>
                    </div>
                `;
            });
        }

        function updateCart(id, change) {
            cart[id] = (cart[id] || 0) + change;
            if (cart[id] <= 0) {
                cart[id] = 0;
                delete cart[id];
            }
            
            // 更新 UI
            document.getElementById(`qty-${id}`).innerText = cart[id] || 0;
            
            // 計算總額
            let total = 0;
            Object.keys(cart).forEach(pid => {
                const p = products.find(x => x.id === pid);
                total += p.price * cart[pid];
            });

            document.getElementById('total-price').innerText = total;
            const bar = document.getElementById('bottom-bar');
            bar.style.transform = total > 0 ? 'translateY(0)' : 'translateY(120px)';
        }

        function openCheckout() {
            const summary = document.getElementById('order-summary');
            summary.innerHTML = '';
            Object.keys(cart).forEach(id => {
                const p = products.find(x => x.id === id);
                summary.innerHTML += `
                    <div class="flex justify-between text-gray-600">
                        <span>${p.name} x ${cart[id]}</span>
                        <span class="font-medium">$${p.price * cart[id]}</span>
                    </div>`;
            });
            document.getElementById('drawer').classList.remove('hidden');
        }

        function closeCheckout() {
            document.getElementById('drawer').classList.add('hidden');
        }

        function finalizeOrder() {
            const name = document.getElementById('user-name').value;
            const adminPhone = "09XXXXXXXX"; // <-- 填寫收件管理員電話
            
            if(!name) { alert("大師，請填寫大名！"); return; }

            const btn = document.getElementById('final-btn');
            btn.disabled = true;
            btn.innerText = "寫入雲端中...";

            const orderStr = Object.keys(cart).map(id => {
                const p = products.find(x => x.id === id);
                return `${p.name}x${cart[id]}`;
            }).join(', ');

            const total = document.getElementById('total-price').innerText;

            // 1. Firebase 儲存
            db.ref('orders').push({
                name: name,
                items: orderStr,
                total: total,
                time: ServerValue.TIMESTAMP
            }).then(() => {
                // 2. 喚起手機簡訊
                const smsBody = `【活力美髮團購】\n姓名：${name}\n明細：${orderStr}\n總額：$${total}`;
                window.location.href = `sms:${adminPhone}?body=${encodeURIComponent(smsBody)}`;
                
                // 3. 重置
                setTimeout(() => {
                    location.reload();
                }, 1000);
            });
        }

        renderProducts();
    </script>
